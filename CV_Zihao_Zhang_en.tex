\Section{Profile}
{\color{Muted}\fontsize{8.5}{11.5}\selectfont
I am a master's student in the School of Computer Science at Fudan University, advised by \href{https://scholar.google.com/citations?user=f3_FP8AAAAAJ\&hl=zh-CN\&authuser=1\&oi=ao}{\textcolor{myblue}{\underline{Prof. Yu-Gang Jiang}}} and \href{https://scholar.google.com/citations?user=7t12hVkAAAAJ\&hl=zh-CN\&authuser=1}{\textcolor{myblue}{\underline{Prof. Zuxuan Wu}}}.

\vspace{0.2em}
My research explores generative visual modeling for \textbf{embodied intelligence}, with the goal of building \textbf{controllable and efficient visual foundations} that help embodied agents understand and predict how scenes evolve over time.

\vspace{0.2em}
My work has been published at leading computer vision and multimedia venues, including \textbf{CVPR}, \textbf{ICCV}, \textbf{ACM MM} and \textbf{SIGGRAPH Asia}.\par
}

\Section{Research Experience}
\ExperienceEntry
  {Apr 2025 \textemdash{} Mar 2026}
  {BiliBili Inc.}
  {Shanghai, China}
  {%
    \Project
      {SPEED: One-Step Pixel Diffusion for High-quality Video Frame Interpolation}
      {Co-first author, ACM MM 2026}
      {\begin{Contributions}
        \item Led the one-step pixel-space diffusion framework and designed a progressive multi-stage DiT, Noise-Update-Only Attention, and Drift-aware Timestep Sampling to learn motion, structure, and appearance from coarse to fine.
        \item Achieved state-of-the-art visual-dynamics prediction: 8.8\% lower LPIPS on SNU-FILM, 63.3\% faster inference, 10.6\% less memory, and up to 51.5\% lower LPIPS on 4K benchmarks.
      \end{Contributions}}
    \Project
      {CT-1: Vision-Language-Camera Models for Camera-Controllable Video Generation}
      {Co-first author, ACM MM 2026}
      {\begin{Contributions}
        \item Co-developed a Vision-Language-Camera model that grounds visual observations and language instructions in SE(3) camera trajectories, enabling spatially aware and temporally stable viewpoint planning.
        \item Contributed to the CT-200K curation pipeline (\textgreater{}47M frames) and controllable video-diffusion system, improving camera-control success by 25.7\% and advancing spatially grounded generative world modeling.
      \end{Contributions}}
  }

      {Third author, ICCV 2025}
      {\begin{Contributions}

        \item Reduced GPU memory by approximately 80\% versus MotionEditor while improving videos with large camera motion and complex backgrounds, enabling efficient motion-conditioned human-scene synthesis.
